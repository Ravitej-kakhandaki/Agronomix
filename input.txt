        <script>
            function addRow(containerId) {
                var container = document.getElementById(containerId);
                var newRow = document.createElement('div');
                newRow.className = containerId.slice(0, -6) + '-row'; // Adjust the class name based on the containerId

                var labels = getLabels(containerId);

                for (var i = 0; i < labels.length; i++) {
                    var label = document.createElement('label');
                    label.textContent = labels[i] + ':';

                    var input = document.createElement('input');
                    input.type = (labels[i].toLowerCase().includes('date')) ? 'date' : 'text';
                    input.name = containerId.slice(0, -6) + '[]'; // Use the same name for each input field in a row
                    input.placeholder = 'Enter ' + labels[i].toLowerCase();

                    newRow.appendChild(label);
                    newRow.appendChild(input);
                }

                var removeButton = document.createElement('button');
                removeButton.type = 'button';
                removeButton.className = 'remove-button';
                removeButton.textContent = 'Remove';
                removeButton.onclick = function () {
                    removeRow(newRow);
                };

                newRow.appendChild(removeButton);
                container.appendChild(newRow);
            }

            function removeRow(row) {
                row.parentNode.removeChild(row);
            }

            function getLabels(containerId) {
                switch (containerId) {
                    case 'farmerFields':
                        return ['Name', 'Contact Info', 'Location', 'Email'];
                    case 'cropFields':
                        return ['Crop Name', 'Crop Type', 'Soil Requirement', 'Ideal Temperature', 'Ideal Humidity'];
                    case 'fieldFields':
                        return ['Field Size', 'Soil Condition', 'Field Name', 'Planting Date'];
                    case 'harvestFields':
                        return ['Harvest Date', 'Quantity Harvested'];
                    case 'pesticidesFields':
                        return ['Pesticide Name', 'Application Date', 'Quantity Used'];
                    case 'irrigationFields':
                        return ['Date of Irrigation', 'Type', 'Duration'];
                    case 'diseaseFields':
                        return ['Disease Name', 'Detection Date', 'Treatment Applied', 'Treatment Date'];
                    case 'soilTestFields':
                        return ['Test Date', 'Soil pH', 'Nutrient Levels', 'Recommended Fertilizers'];
                    case 'qualityFields':
                        return ['Sample Date', 'Quality Rating', 'Weather Conditions'];
                    case 'inspectorFields':
                        return ['First and Last Name', 'Contact Number', 'Email'];
                    case 'marketPriceFields':
                        return ['Date', 'Market Name', 'Price per Unit'];
                    default:
                        return [];
                }
            }
        </script>
